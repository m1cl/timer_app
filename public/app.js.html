1 const TimersDashboard = React.createClass({
  2     getInitialState: function () {
  3         return {
  4             timers: [
  5                 {
  6                     title: 'Practice squat',
  7                     project: 'Gym Chores',
  8                     id: uuid.v4(),
  9                     elapsed: 5456099,
 10                     runningSince: Date.now(),
 11                 },
 12                 {
 13                     title: 'Bake squash',
 14                     project: 'Kitchen Chores',
 15                     id: uuid.v4(),
 16                     elapsed: 1273998,
 17                     runningSince: null,
 18                 },
 19             ],
 20         };
 21     },
 22     handleCreateFormSubmit: function (timer) {
 23         this.createTimer(timer);
 24     },
 25     handleEditFormSubmit: function (attrs) {
 26         this.updateTimer(attrs);
 27     },
 28     createTimer: function (timer) {
 29         const t = helpers.newTimer(timer);
 30         this.setState ({
 31             timers: this.state.timers.concat(t),
 32         });
 33     },
 34     updateTimer: function (attrs) {
 35         this.setState({
 36             timers: this.state.timers.map((timer) => {
 37                 if(timer.id === attrs.id) {
 38                     return Object.assign({}, timer, {
 39                         title: attrs.title,
 40                         project: attrs.project,
 41                     });
 42                 } else {
 43                     return timer;
 44                 }
 45             }),
 46         });
 47     },
 48     render: function () {
 49         return (
 50             <div className='ui three column centered grid'>
 51                 <div className='column'>
 52                     <EditableTimerList
 53                         timers={this.state.timers} 
 54                         onFormSubmit={this.handleEditFormSubmit}
 55                     />
 56                     <ToggleableTimerForm 
 57                         onFormSubmit={this.handleCreateFormSubmit}
 58                     />
 59                 </div>
 60             </div>
 61         );
 62     },
 63 });

 85 const EditableTimer = React.createClass({
 86     getInitialState: function () {
 87         return {
 88             editFormOpen: false,
 89         };
 90     },
 91     handleEditClick: function () {
 92         this.openForm();
 93     },
 94     handleFormClose: function () {
 95         this.closeForm();
 96     },
 97     handleSubmit: function (timer) {
 98         this.props.onFormSubmit(timer);
 99         this.closeForm();
100     },
101     closeForm: function () {
102         this.setState( {editFormOpen: false} );
103     },
104     openForm: function () {
105         this.setState ({editFormOpen: true});
106     },
107     render: function () {
108         if (this.state.editFormOpen) {
109             return (
110                 <TimerForm
111                     id={this.props.id}
112                     title={this.props.title}
113                     project={this.props.project}
114                     runningSince={this.props.runningSince}
115                     onFormSubmit={this.handleSubmit}
116                     onFormClose={this.handleFormClose}
117                 />
118         );
119         }else {
120             return (
121                 <Timer
122                     id={this.props.id}
123                     title={this.props.title}
124                     project={this.props.project}
125                     elapsed={this.props.elapsed}
126                     runningSince={this.props.runningSince}
127                     onEditClick={this.handleEditClick}
128                 />
129             );
130         }
131     },
132 });

175 const ToggleableTimerForm = React.createClass( {
176     getInitialState: function () {
177         return {
178             isOpen: false,
179         };
180     },
181     handleFormOpen: function () {
182         this.setState({isOpen: true});
183     },
184     handleFormClose: function () {
185         this.setState({ isOpen: false })
186     },
187     handleFormSubmit: function (timer) {
188         this.props.onFormSubmit(timer);
189         this.setState({ isOpen: false })
190     },
191     render: function () {
192         if (this.state.isOpen) {
193             return (
194                 <TimerForm 
195                     onFormSubmit={this.handleFormSubmit}
196                     onFormClose={this.handleFormClose}
197                 />
198             );
199         } else {
200             return (
201                 <div className='ui basic content center aligned segment'>
202                     <button 
203                         className='ui basic button icon '
204                         onClick={this.handleFormOpen}
205                     >
206                         <i className='plus icon'></i>
207                     </button>
208                 </div>
209             );
210         }
211     },
212 } );
